Restaurant : RestaurantID, Address
Food : FoodID, FoodName, UnitPrice
Groups : GroupID, GroupName
Employee : EmployeeID, GroupID, Name, Address
OrderDetail : OrdersID, RestaurantID, FoodID, Quantity
Orders : OrderID, GroupID, OrderDate, RestaurantID

# 建立資料庫
CREATE DATABASE cyDinbendon;
USE cyDinbendon;

# 建立資料表
CREATE TABLE Restaurant(
    RestaurantID int AUTO_INCREMENT PRIMARY KEY,   
   	Address varchar(30)
);

CREATE TABLE Food(
    FoodID int AUTO_INCREMENT PRIMARY KEY,
    FoodName varchar(15) not null,
    UnitPrice int not null
);

CREATE TABLE Groups(
    GroupID int AUTO_INCREMENT PRIMARY KEY,
    GroupName varchar(15) not null
);

CREATE TABLE Employee(
    EmployeeID int AUTO_INCREMENT PRIMARY KEY,
    GroupID int not null,
    Name varchar(10) not null,
    Address varchar(30)
);

CREATE TABLE OrderDetail(
    OrderID int not null,
    RestaurantID int not null,
    FoodID int not null,
    Quantity int not null    
);

CREATE TABLE Orders(
    OrderID int AUTO_INCREMENT PRIMARY KEY,
    GroupID int not null,
    OrderDate date not null,
    RestaurantID int not null
);

# 建立關聯
ALTER TABLE Employee
ADD CONSTRAINT fk_emp_grp
FOREIGN KEY (GroupID) REFERENCES Groups(GroupID);

ALTER TABLE Orders
ADD CONSTRAINT fk_odr_grp
FOREIGN KEY (GroupID) REFERENCES Groups(GroupID);

ALTER TABLE Orders
ADD CONSTRAINT fk_odr_rest
FOREIGN KEY (RestaurantID) REFERENCES Restaurant(RestaurantID);

ALTER TABLE Orderdetail
ADD CONSTRAINT fk_odrdtl_odr
FOREIGN KEY (OrderID) REFERENCES Orders(OrderID);

ALTER TABLE Orderdetail
ADD CONSTRAINT fk_odrdtl_rest
FOREIGN KEY (RestaurantID) REFERENCES Restaurant(RestaurantID);

ALTER TABLE Orderdetail
ADD CONSTRAINT fk_odrdtl_fd
FOREIGN KEY (FoodID) REFERENCES Food(FoodID);
